def format_(df):
    c=0
    b = []
    start, prev = 0, 0
    for i in df.index:
        if df.iloc[i]['actual_value'] == 0:
            c+=1
        elif df.iloc[i]['actual_value'] == 1:
    #         b.append(0)
            prev=c
            c=0
            if(prev == 0):
                b.append(0)
            else:
                rt = [l for l in range(prev,0,-1)]
                for n in rt:
                    b.append(n)

        if i == len(df.index)-1:

            for n in rt:
                b.append(n)
    return b

Breakdown Prediction¶

I = Instance Data
a = smooth window
b = threshold value
T = actual RUL
In [353]:
# Takes in four parameters I,a,b,T
def predict(I,a,b,T):
    AOI = []
    mm = len(I)
    val = int(mm/a)+1
    t=[]
    for i in range(1,val):
        start = ((i-1)*a)+1
        nn=0
        fin = i*a
        x = I[start:fin]
        rul = np.array(T[start:fin]).mean()
        for i in x:
            if i==0:
                nn +=1
        aoi = 1-(nn/a)
#         print('start',start)
#         print('fin',fin)
        AOI.append(aoi)
        t.append(rul)
    z = []
    for i in range(0,len(AOI)):
        if AOI[i]>=b:
            z.append(1)
        else:
            z.append(0)    
    
    return np.array(np.array(z)*np.array(t)).sum()
